def nullpreprocessing(data,column,col) :
    null = data[data[column].isnull()]
    notnull = data[data[column].notnull()]
    series = notnull[column]
    null.drop(col,axis = 1,inplace = True)
    notnull.drop(col,axis = 1,inplace = True)
    rf = RandomForestRegressor()
    rf.fit(notnull,series)
    predicted = rf.predict(null)
    data[column][data[column].isnull()] = predicted
    return data
